void %main.type_name%_trim_leading_zeroes(struct %main.base_type%*a)
{
    for (size_t i = a->coefficient_count; i-- > 0;)
    {
        if (%coefficient.base_type_name%_equals(a->coefficients[i], %coefficient.additive_identity%))
        {
            a->coefficient_count -= 1;
        }
        else
        {
            return;
        }
    }
}
